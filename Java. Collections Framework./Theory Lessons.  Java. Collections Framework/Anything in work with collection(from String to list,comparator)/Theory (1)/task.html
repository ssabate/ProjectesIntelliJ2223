List<Integer> lst = new ArrayList<>();
//in ascending order
lst.sort((o1, o2) -> o1.compareTo(o2)); //lst.sort(Integer::compareTo);
Collections.sort(lst,Comparator.naturalOrder());
Collections.sort(lst);

//in descending order
lst.sort((o1, o2) -> o2.compareTo(o1));
Collections.sort(lst,Comparator.reverseOrder());Collections.sort(lst,(o1, o2) -> o2.compareTo(o1));

//with Stream API
lst.stream().filter(i -> i%2==0).sorted(Comparator.reverseOrder()).forEach(System.out::println);
